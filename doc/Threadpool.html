def initialize(size)
  @queue = Queue.new
  @pool = []
  size.times do |i|
    @pool << Thread.new do
      loop do
        job, args = @queue.pop
        begin
          job.call(*args)
        rescue Exception => e
          puts "Thread #{Thread.current} error: #{e} #{e.message} #{e.backtrace.join("\n")}"
        end
      end
    end
  end
end
