def initialize(options={})
  @options = options
  @bind = @options[:bind] || '0.0.0.0'
  @port = (@options[:port] || 8125) + 2
  Batsd::Server.config = @options.merge(port: @port)
end

def run
  EventMachine.threadpool_size = 100
